import { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
    return {
        rules: {
            userAgent: "*",
            allow: "/",
            disallow: ["/api/", "/sign-in", "/sign-up", "/settings/"],
        },
        sitemap: "https://astraeus-livid.vercel.app/sitemap.xml",
    };
}
